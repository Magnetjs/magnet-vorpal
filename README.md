### Usage
Basic
command.js
```
import magnet from 'magnet-core';
import Config from 'magnet-config';
import Logger from 'magnet-bunyan';
import Vorpal from 'magnet-vorpal';

let app = await magnet([
  [Koa, Config],
  Logger,
  Vorpal
]);
```
